#include <bits/stdc++.h>
using namespace std ;



int main(){
    int profit[100];
    int weight[100];

    int n;
    int W;
    cout<<"Enter the number of items\t";
    cin>>n;
    cout<<"Enter the capacity of the bag \t";
    cin>>W;
    
    for(int i=1;i<=n;i++){
        cout<<"Enter the weight & profit of item "<<i;
        cin>>weight[i];
        cin>>profit[i];
    }
    int dp [100][100];

    for(int i=0;i<=n;i++){
        dp[i][0]=0;
    }
    for(int i=0;i<=W;i++){
        dp[0][i]=0;
    }
    for(int i=1;i<=n;i++){
       for (int w=1;w<=W;w++){
        if(weight[i]<=w){
            dp[i][w]=max(dp[i-1][w],profit[i]+dp[i-1][w-weight[i]]);
           
        }
        else
        {
           dp[i][w] = dp[i-1][w];
        }
        
       }

    }

    



    cout<<dp[n][W];
}
