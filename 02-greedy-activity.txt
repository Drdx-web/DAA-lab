#include <bits/stdc++.h>
using namespace std;

struct Activity {
    int s, f;
};

bool cmp(Activity a, Activity b) {
    return a.f < b.f;
}

int main() {

    int n;
    cin >> n;

    vector<Activity> a(n);

    for(int i = 0; i < n; i++)
        cin >> a[i].s >> a[i].f;

    sort(a.begin(), a.end(), cmp);

    cout << "Selected Activities:\n";

    int count = 1, last = a[0].f;

    cout << "(" << a[0].s << "," << a[0].f << ")\n";

    for(int i = 1; i < n; i++) {

        if(a[i].s >= last) {
            cout << "(" << a[i].s << "," << a[i].f << ")\n";
            last = a[i].f;
            count++;
        }
    }

    cout << "Maximum Activities = " << count;

    return 0;
}

/*
Sample Input:

6
1 2
3 4
0 6
5 7
8 9
5 9

Sample Output:

Selected Activities:
(1,2)
(3,4)
(5,7)
(8,9)

Maximum Activities = 4
*/
