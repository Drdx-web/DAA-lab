#include <bits/stdc++.h>
using namespace std;

struct Item {
    int v, w;
};

bool cmp(Item a, Item b) {
    return (double)a.v / a.w > (double)b.v / b.w;
}

int main() {

    int n, W;
    cin >> n;

    vector<Item> a(n);

    for(int i = 0; i < n; i++)
        cin >> a[i].v >> a[i].w;

    cin >> W;

    sort(a.begin(), a.end(), cmp);

    double ans = 0;

    for(auto i : a) {

        if(W >= i.w) {
            W -= i.w;
            ans += i.v;
        }
        else {
            ans += i.v * ((double)W / i.w);
            break;
        }
    }

    cout << "Maximum Value = " << ans;

    return 0;
}

/*
Sample Input:

3
60 10
100 20
120 30
50

Sample Output:

Maximum Value = 240
*/
