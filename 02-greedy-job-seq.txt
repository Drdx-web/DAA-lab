#include <bits/stdc++.h>
using namespace std;

struct Job {
    int id, d, p;
};

bool cmp(Job a, Job b) {
    return a.p > b.p;
}

int main() {

    int n;
    cin >> n;

    vector<Job> a(n);

    for(int i = 0; i < n; i++)
        cin >> a[i].id >> a[i].d >> a[i].p;

    sort(a.begin(), a.end(), cmp);

    int mx = 0, profit = 0;

    for(auto i : a)
        mx = max(mx, i.d);

    vector<int> slot(mx + 1, -1);

    cout << "Selected Jobs:\n";

    for(auto i : a) {

        for(int j = i.d; j > 0; j--) {

            if(slot[j] == -1) {

                slot[j] = i.id;
                profit += i.p;

                cout << "Job " << i.id
                     << " at slot " << j << endl;

                break;
            }
        }
    }

    cout << "Total Profit = " << profit;

    return 0;
}

/*
Sample Input:

5
1 2 100
2 1 19
3 2 27
4 1 25
5 3 15

Sample Output:

Selected Jobs:
Job 1 at slot 2
Job 3 at slot 1
Job 5 at slot 3

Total Profit = 142
*/
