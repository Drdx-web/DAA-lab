#include <bits/stdc++.h>
using namespace std;

int main() {

    int n, V;
    cin >> n;

    vector<int> coin(n);

    for(int i = 0; i < n; i++)
        cin >> coin[i];

    cin >> V;

    vector<int> dp(V + 1, INT_MAX);

    dp[0] = 0;

    for(int i = 1; i <= V; i++) {

        for(int c : coin) {

            if(c <= i && dp[i - c] != INT_MAX)
                dp[i] = min(dp[i], dp[i - c] + 1);
        }
    }

    if(dp[V] == INT_MAX)
        cout << "Not Possible";
    else
        cout << "Minimum Coins = " << dp[V];

    return 0;
}

/*
Sample Input:

3
1 2 5
11

Sample Output:

Minimum Coins = 3
*/
