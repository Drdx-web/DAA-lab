#include <bits/stdc++.h>
using namespace std;

int coinChangeDP(vector<int> &coins, int V) {
    vector<int> dp(V + 1, INT_MAX);

    dp[0] = 0; // Base case

    // Build DP table
    for (int i = 1; i <= V; i++) {
        for (int coin : coins) {
            if (coin <= i && dp[i - coin] != INT_MAX) {
                dp[i] = min(dp[i], 1 + dp[i - coin]);
            }
        }
    }

    // If not possible
    if (dp[V] == INT_MAX)
        return -1;

    return dp[V];
}

int main() {
    int n, V;

    cout << "Enter number of denominations: ";
    cin >> n;

    vector<int> coins(n);

    cout << "Enter coin denominations:\n";
    for (int i = 0; i < n; i++)
        cin >> coins[i];

    cout << "Enter amount: ";
    cin >> V;

    clock_t start = clock();

    int result = coinChangeDP(coins, V);

    clock_t end = clock();

    if (result == -1)
        cout << "\nNot possible to form amount";
    else
        cout << "\nMinimum coins required = " << result;

    double time_taken = double(end - start) / CLOCKS_PER_SEC;
    cout << "\nTime taken: " << time_taken << " seconds\n";

    return 0;
}
