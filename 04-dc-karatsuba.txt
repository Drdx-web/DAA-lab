#include <bits/stdc++.h>
using namespace std;

string add(string a, string b) {

    if(a.size() < b.size()) swap(a, b);

    int carry = 0, d = a.size() - b.size();
    string r = "";

    for(int i = b.size() - 1; i >= 0; i--) {
        int s = (a[i + d] - '0') + (b[i] - '0') + carry;
        r += (s % 10) + '0';
        carry = s / 10;
    }

    for(int i = d - 1; i >= 0; i--) {
        int s = (a[i] - '0') + carry;
        r += (s % 10) + '0';
        carry = s / 10;
    }

    if(carry) r += carry + '0';

    reverse(r.begin(), r.end());
    return r;
}

string sub(string a, string b) {

    int d = a.size() - b.size(), borrow = 0;
    string r = "";

    for(int i = b.size() - 1; i >= 0; i--) {

        int s = (a[i + d] - '0') - (b[i] - '0') - borrow;

        if(s < 0) s += 10, borrow = 1;
        else borrow = 0;

        r += s + '0';
    }

    for(int i = d - 1; i >= 0; i--) {

        int s = (a[i] - '0') - borrow;

        if(s < 0) s += 10, borrow = 1;
        else borrow = 0;

        r += s + '0';
    }

    while(r.size() > 1 && r.back() == '0')
        r.pop_back();

    reverse(r.begin(), r.end());
    return r;
}

string karatsuba(string x, string y) {

    while(x.size() < y.size()) x = "0" + x;
    while(y.size() < x.size()) y = "0" + y;

    int n = x.size();

    if(n == 1)
        return to_string((x[0]-'0') * (y[0]-'0'));

    int m = n / 2;

    string a = x.substr(0, n - m);
    string b = x.substr(n - m);
    string c = y.substr(0, n - m);
    string d = y.substr(n - m);

    string p1 = karatsuba(a, c);
    string p2 = karatsuba(b, d);
    string p3 = karatsuba(add(a,b), add(c,d));

    string mid = sub(sub(p3, p1), p2);

    p1 += string(2 * m, '0');
    mid += string(m, '0');

    return add(add(p1, mid), p2);
}

int main() {

    string a, b;
    cin >> a >> b;

    cout << "Product = " << karatsuba(a, b);

    return 0;
}
