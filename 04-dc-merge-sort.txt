#include <bits/stdc++.h>
using namespace std;

void merge(vector<int> &a, int l, int m, int r) {

    vector<int> t;
    int i = l, j = m + 1;

    while(i <= m && j <= r)
        t.push_back(a[i] <= a[j] ? a[i++] : a[j++]);

    while(i <= m) t.push_back(a[i++]);
    while(j <= r) t.push_back(a[j++]);

    for(int k = l; k <= r; k++)
        a[k] = t[k - l];
}

void mergeSort(vector<int> &a, int l, int r) {

    if(l < r) {
        int m = (l + r) / 2;

        mergeSort(a, l, m);
        mergeSort(a, m + 1, r);

        merge(a, l, m, r);
    }
}

int main() {

    int n;
    cin >> n;

    vector<int> a(n);

    for(int i = 0; i < n; i++)
        cin >> a[i];

    mergeSort(a, 0, n - 1);

    cout << "Sorted Array:\n";

    for(int x : a)
        cout << x << " ";

    return 0;
}

/*
Sample Input:

6
12 11 13 5 6 7

Sample Output:

Sorted Array:
5 6 7 11 12 13
*/
