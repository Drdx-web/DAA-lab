#include <bits/stdc++.h>
using namespace std;

int main() {

    int a[] = {12, 45, 7, 89, 23, 56, 1, 99};
    int n = sizeof(a) / sizeof(a[0]);

    int mn, mx, i;

    if(n % 2 == 0) {

        if(a[0] < a[1])
            mn = a[0], mx = a[1];
        else
            mn = a[1], mx = a[0];

        i = 2;
    }
    else {
        mn = mx = a[0];
        i = 1;
    }

    while(i < n - 1) {

        int lmn, lmx;

        if(a[i] < a[i + 1])
            lmn = a[i], lmx = a[i + 1];
        else
            lmn = a[i + 1], lmx = a[i];

        mn = min(mn, lmn);
        mx = max(mx, lmx);

        i += 2;
    }

    cout << "Minimum = " << mn << endl;
    cout << "Maximum = " << mx;

    return 0;
}

/*
Sample Output:

Minimum = 1
Maximum = 99
*/
