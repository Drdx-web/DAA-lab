#include <bits/stdc++.h>
using namespace std;

int median(vector<int> &a, int l, int n) {
    sort(a.begin() + l, a.begin() + l + n);
    return a[l + n / 2];
}

int mom(vector<int> &a, int l, int r) {

    int n = r - l + 1;

    if(n <= 5)
        return median(a, l, n);

    vector<int> m;

    for(int i = 0; i < n / 5; i++)
        m.push_back(median(a, l + i * 5, 5));

    if(n % 5)
        m.push_back(median(a, l + (n / 5) * 5, n % 5));

    return mom(m, 0, m.size() - 1);
}

int partition(vector<int> &a, int l, int r, int p) {

    int idx;

    for(int i = l; i <= r; i++) {
        if(a[i] == p) {
            idx = i;
            break;
        }
    }

    swap(a[idx], a[r]);

    int i = l;

    for(int j = l; j < r; j++) {

        if(a[j] <= p)
            swap(a[i++], a[j]);
    }

    swap(a[i], a[r]);

    return i;
}

void quickSort(vector<int> &a, int l, int r) {

    if(l < r) {

        int p = mom(a, l, r);

        int pos = partition(a, l, r, p);

        quickSort(a, l, pos - 1);
        quickSort(a, pos + 1, r);
    }
}

int main() {

    int n;
    cin >> n;

    vector<int> a(n);

    for(int i = 0; i < n; i++)
        cin >> a[i];

    quickSort(a, 0, n - 1);

    cout << "Sorted Array:\n";

    for(int x : a)
        cout << x << " ";

    return 0;
}

/*
Sample Input:

6
12 4 5 3 8 7

Sample Output:

Sorted Array:
3 4 5 7 8 12
*/
