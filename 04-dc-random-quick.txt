#include <bits/stdc++.h>
using namespace std;

int partition(vector<int> &a, int l, int h) {

    int p = a[h], i = l - 1;

    for(int j = l; j < h; j++) {

        if(a[j] < p)
            swap(a[++i], a[j]);
    }

    swap(a[i + 1], a[h]);

    return i + 1;
}

int randPartition(vector<int> &a, int l, int h) {

    int r = l + rand() % (h - l + 1);

    swap(a[r], a[h]);

    return partition(a, l, h);
}

void quickSort(vector<int> &a, int l, int h) {

    if(l < h) {

        int pi = randPartition(a, l, h);

        quickSort(a, l, pi - 1);
        quickSort(a, pi + 1, h);
    }
}

int main() {

    int n;
    cin >> n;

    vector<int> a(n);

    for(int i = 0; i < n; i++)
        cin >> a[i];

    srand(time(0));

    quickSort(a, 0, n - 1);

    cout << "Sorted Array:\n";

    for(int x : a)
        cout << x << " ";

    return 0;
}

/*
Sample Input:

6
12 4 5 3 8 7

Sample Output:

Sorted Array:
3 4 5 7 8 12
*/
