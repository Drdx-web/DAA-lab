#include <bits/stdc++.h>
using namespace std;

int sum(int freq[], int i, int j)
{
    int s = 0;

    for(int k=i; k<=j; k++)
    {
        s = s + freq[k];
    }

    return s;
}

int main()
{
    int n;

    int key[100];

    int freq[100];

    cout << "Enter number of keys: ";
    cin >> n;

    cout << "Enter keys: ";

    for(int i=0; i<n; i++)
    {
        cin >> key[i];
    }

    cout << "Enter frequencies: ";

    for(int i=0; i<n; i++)
    {
        cin >> freq[i];
    }

    int cost[100][100];

    for(int i=0; i<n; i++)
    {
        cost[i][i] = freq[i];
    }

    for(int len=2; len<=n; len++)
    {
        for(int i=0; i<=n-len; i++)
        {
            int j = i + len - 1;

            cost[i][j] = INT_MAX;

            int fsum = sum(freq, i, j);

            for(int r=i; r<=j; r++)
            {
                int left;

                int right;

                if(r > i)
                {
                    left = cost[i][r-1];
                }
                else
                {
                    left = 0;
                }

                if(r < j)
                {
                    right = cost[r+1][j];
                }
                else
                {
                    right = 0;
                }

                int total =
                left + right + fsum;

                cost[i][j] =
                min(cost[i][j], total);
            }
        }
    }

    cout << "Optimal Cost = "
         << cost[0][n-1];

    return 0;
}
/*
Sample Input:

3
0.15 0.10 0.05
0.05 0.10 0.05 0.05

Sample Output:

Minimum Cost = 1.25
*/
