#include <bits/stdc++.h>
using namespace std;

int main() {

    int n;
    cin >> n;

    vector<float> p(n + 1), q(n + 1);

    for(int i = 1; i <= n; i++)
        cin >> p[i];

    for(int i = 0; i <= n; i++)
        cin >> q[i];

    vector<vector<float>> e(n + 2, vector<float>(n + 1, 0));
    vector<vector<float>> w(n + 2, vector<float>(n + 1, 0));

    for(int i = 1; i <= n + 1; i++) {
        e[i][i - 1] = q[i - 1];
        w[i][i - 1] = q[i - 1];
    }

    for(int l = 1; l <= n; l++) {

        for(int i = 1; i <= n - l + 1; i++) {

            int j = i + l - 1;

            e[i][j] = INT_MAX;
            w[i][j] = w[i][j - 1] + p[j] + q[j];

            for(int r = i; r <= j; r++) {

                float cost = e[i][r - 1] +
                             e[r + 1][j] +
                             w[i][j];

                e[i][j] = min(e[i][j], cost);
            }
        }
    }

    cout << "Minimum Cost = " << e[1][n];

    return 0;
}

/*
Sample Input:

3
0.15 0.10 0.05
0.05 0.10 0.05 0.05

Sample Output:

Minimum Cost = 1.25
*/
