#include <bits/stdc++.h>
using namespace std;

struct activity{
    int start;
    int finish;
};

bool comp(activity a,activity b){
    return a.finish<b.finish;
}


int main(){
    int n;
    activity arr [100];
    cout<<"Enter the number of activities ";
    cin>>n;
    for(int i=0;i<n;i++){
        cout<<"Enter the start & end time for activity - "<<i+1<<endl;
        cin>>arr[i].start;
        cin>>arr[i].finish;
    }
    sort(arr,arr+n,comp);
    cout<<"Selected Activities "<<endl;
    cout<<arr[0].start<<"\t"<<arr[0].finish<<endl;
    int lastact=arr[0].finish;
    
    for(int i=1;i<n;i++){
        if (arr[i].start>=lastact){
            cout<<arr[i].start<<"\t"<<arr[i].finish<<endl;
            lastact=arr[i].finish;
        }

    }
    return 0;

}

Selected Activities:
