#include <bits/stdc++.h>
using namespace std;

int main() {

    int n, V;
    cin >> n;

    vector<int> coin(n);

    for(int i = 0; i < n; i++)
        cin >> coin[i];

    cin >> V;

    sort(coin.rbegin(), coin.rend());

    int count = 0;

    cout << "Coins Used:\n";

    for(int c : coin) {

        while(V >= c) {
            cout << c << " ";
            V -= c;
            count++;
        }
    }

    cout << "\nMinimum Coins = " << count;

    return 0;
}

/*
Sample Input:

4
1 2 5 10
18

Sample Output:

Coins Used:
10 5 2 1

Minimum Coins = 4
*/
