#include <bits/stdc++.h>
using namespace std;

struct item
{
    int profit ;
    int weight ;
};
bool comp(item a,item b){
    double r1= (double) a.profit/a.weight;
    double r2=(double) b.profit/b.weight;
    return r1>r2;
}

int main(){
    int n;
    int w;
    double p=0;
    item arr[100];
    cout<<"Enter the max weight of the bag";
    cin>>w;
    cout<<"Enter the no of items";
    cin>>n;
    for (int i=0;i<n;i++){
        cout<<"Enter the profit and weight of item\t"<<i+1<<endl;
        cin>>arr[i].profit;
        cin>>arr[i].weight;
    }
    sort(arr,arr+n,comp);
    int r=w;//to track remaining weight
    for (int i=0;i<n;i++){
        if(r>=arr[i].weight){
            r=r-arr[i].weight;
            p=p+arr[i].profit;
        }
        else if(r==0){
            break;
        }
        else{
            
            p=p+((double)arr[i].profit/arr[i].weight)*r;
            break;
        }

    }
    cout<<"Max profit can be\t"<<p;
}

/*
Sample Input:

3
60 10
100 20
120 30
50

Sample Output:

Maximum Value = 240
