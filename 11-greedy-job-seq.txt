#include <bits/stdc++.h>
using namespace std;

struct Job
{
   char id;
   int deadline;
   int profit;
};

bool comp(Job a,Job b){
    return a.profit>b.profit;
}




int main(){
    Job arr[100];
    int n;
    int p=0;
    cout<<"Enter the number of jobs ";
    cin>>n;
    for(int i=0;i<n;i++){
        cout<<"Enter the id , deadline & profit for "<<i+1<<"st job ";
        cin>>arr[i].id;
        cin>>arr[i].deadline;
        cin>>arr[i].profit;
    }
    sort(arr,arr+n,comp);
    int slot [100];
    for(int i=0;i<100;i++){
        slot[i]=-1;
    }
    cout<<"Selected jobs are "<<endl;
    for(int i=0;i<n;i++){
        for(int j=arr[i].deadline-1;j>=0;j--){
            if(slot[j]==-1){
                slot[j]=i;
                cout<<arr[i].id<<"\t";
                p=p+arr[i].profit;
                break;
            }
        }
    }

    
    cout<<"Total profit = "<<p<<endl;
    return 0;
}
/*
Sample Input:

5
1 2 100
2 1 19
3 2 27
4 1 25
5 3 15

Sample Output:

Selected Jobs:
Job 1 at slot 2
Job 3 at slot 1
Job 5 at slot 3

Total Profit = 142
