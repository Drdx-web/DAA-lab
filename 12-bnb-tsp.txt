#include <bits/stdc++.h>
using namespace std;

int graph[10][10];
int visited[10];
int n;
int ans=INT_MAX;

void tsp(int currCity,int count ,int currCost){
    if(count==n&&graph[currCity][0]){
        ans=min(ans,currCost+graph[currCity][0]);
        return;
    }
    for(int i=0;i<n;i++){
        if(visited[i]==0&&currCost+graph[currCity][i]<ans){
            visited[i]=1;
            tsp(i,count+1,currCost+graph[currCity][i]);
            visited[i]=0;
        }
    }
}




int main(){
    cout<<"Enter the number of cities\t";
    cin>>n;
    cout<<"Enter the cost matrix";
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cin>>graph[i][j];
        }
    }
    for(int i=0;i<n;i++){
        visited[i]=0;

    }
    visited[0]=1;
    tsp(0,1,0);
    cout<<"total cost = "<<ans;

    return 0;
    
}
/*
Sample Input:

4
0 10 15 20
10 0 35 25
15 35 0 30
20 25 30 0

Sample Output:

Minimum Cost = 80
Path: 0 1 3 2 0
