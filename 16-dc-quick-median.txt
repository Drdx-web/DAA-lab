#include <bits/stdc++.h>
using namespace std;

int partition (int arr[],int low,int high){
    int i=low-1;
    int j=low;
    int p=arr[high];
    for(j;j<high;j++){
        if(arr[j]<p){
            i++;
            swap(arr[i],arr[j]);
        }

    }
    swap(arr[i+1],arr[high]);
    return i+1;
}

void qickSort(int arr[],int low,int high){
    if (low<high)
    {
        int p=partition(arr,low,high);
    qickSort(arr,low,p-1);
    qickSort(arr,p+1,high);
    }
    
    
}

int main(){
    int n;
    int arr[100];
    cout<<"Enter the number of elements"<<endl;
    cin>>n;
    cout<<"Enter the elements "<<endl;
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    qickSort(arr,0,n-1);
    for(int i=0;i<n;i++){
        cout<<arr[i];
    }
}
