#include <bits/stdc++.h>
using namespace std;

bool isSafe (int board[10][10],int row,int col,int n){
    for(int i=0;i<row;i++){
        if (board[i][col]==1){
            return false;
        }
    }
    for(int i=row,j=col;i>=0&&j>=0;i--,j--){
        if (board[i][j]==1){
            return false;
        }
    }
    for(int i=row,j=col;i>=0&&j<=n;i--,j++){
        if (board[i][j]==1){
            return false;
        }
    }
    return true;
}
bool solve(int board[10][10],int row,int n){
    if (row==n){
        return true;
    }
    for (int col=0;col<n;col++){
        if(isSafe(board,row,col,n)){
            board[row][col]=1;
            if(solve(board,row+1,n)){
                return true;
            }
            board[row][col]=0;
        }

    }
    return false;
}

int main(){
    int board[10][10]={0};
    int n;
    cout<<"Enter the number of queens"<<endl;
    cin>>n;
    int row =0;
    solve(board,row,n);
    for (int i=0;i<n;i++){
        for (int j=0;j<n;j++){
            cout<<board[i][j];
        }
        cout<<endl;
    }
    return 0;
}
