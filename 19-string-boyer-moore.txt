#include <bits/stdc++.h>
using namespace std;

#define SIZE 256

int main() {

    string t, p;
    cin >> t >> p;

    int n = t.size(), m = p.size(), bad[SIZE];

    for(int i = 0; i < SIZE; i++)
        bad[i] = -1;

    for(int i = 0; i < m; i++)
        bad[p[i]] = i;

    cout << "Pattern Found At: ";

    int s = 0;

    while(s <= n - m) {

        int j = m - 1;

        while(j >= 0 && p[j] == t[s + j])
            j--;

        if(j < 0) {
            cout << s << " ";
            s += (s + m < n) ? m - bad[t[s + m]] : 1;
        }
        else
            s += max(1, j - bad[t[s + j]]);
    }

    return 0;
}

/*
Sample Input:

ABAAABCD
ABC

Sample Output:

Pattern Found At: 4
*/
