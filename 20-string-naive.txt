#include <bits/stdc++.h>
using namespace std ;





int main(){
    string text;
    string pattern;
    cout<<"Enter text\t";
    cin>>text;
    cout<<"Enter the pattern\t";
    cin>>pattern;
    int n=text.length();
    int m=pattern.length();

    for(int i=0;i<=n-m;i++){
        int j;
        for(j=0;j<m;j++){
            if (text[i+j]!=pattern[j]){
                break;
            }
        }
        if(j==m){
            cout<<"Pattern found at index\t"<<i<<endl;
        }
    }
    return 0;
}
/*
Sample Input:

AABAACAADAABAABA
AABA

Sample Output:

Pattern Found At: 0 9 12
*/
