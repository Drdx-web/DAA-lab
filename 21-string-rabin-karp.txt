#include <bits/stdc++.h>
using namespace std;

#define d 256

int main() {

    string t, p;
    cin >> t >> p;

    int q = 101;
    int n = t.size(), m = p.size();
    int ph = 0, th = 0, h = 1;

    for(int i = 0; i < m - 1; i++)
        h = (h * d) % q;

    for(int i = 0; i < m; i++) {
        ph = (d * ph + p[i]) % q;
        th = (d * th + t[i]) % q;
    }

    cout << "Pattern Found At: ";

    for(int i = 0; i <= n - m; i++) {

        if(ph == th) {

            int j;

            for(j = 0; j < m; j++)
                if(t[i + j] != p[j]) break;

            if(j == m)
                cout << i << " ";
        }

        if(i < n - m) {

            th = (d * (th - t[i] * h) + t[i + m]) % q;

            if(th < 0)
                th += q;
        }
    }

    return 0;
}

/*
Sample Input:

AABAACAADAABAABA
AABA

Sample Output:

Pattern Found At: 0 9 12
*/
